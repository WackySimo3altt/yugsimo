let clicks = 0; let clickValue = 1; let upgradeCost = 10; let rebirthCost = 1000; let isSpacePressed = false; const clicksDisplay = document.getElementById('clicks'); const clickButton = document.getElementById('clickButton'); const upgradeButton = document.getElementById('upgradeButton'); const upgradeCostDisplay = document.getElementById('upgradeCost'); const rebirthButton = document.getElementById('rebirthButton'); const rebirthCostDisplay = document.getElementById('rebirthCost'); clickButton.addEventListener('click', () => { clicks += clickValue; updateDisplay(); }); upgradeButton.addEventListener('click', () => { if (clicks >= upgradeCost) { clicks -= upgradeCost; clickValue *= 2; upgradeCost *= 2; updateDisplay(); } }); rebirthButton.addEventListener('click', () => { if (clicks >= rebirthCost) { clicks = 0; clickValue = 1; upgradeCost = 10; rebirthCost *= 2; updateDisplay(); } }); document.addEventListener('keydown', (event) => { if (event.code === 'Space') { isSpacePressed = true; } }); document.addEventListener('keyup', (event) => { if (event.code === 'Space') { isSpacePressed = false; } }); function updateDisplay() { clicksDisplay.textContent = clicks; upgradeCostDisplay.textContent = upgradeCost; rebirthCostDisplay.textContent = rebirthCost; } function autoClick() { if (isSpacePressed) { clicks += clickValue; updateDisplay(); } requestAnimationFrame(autoClick); } updateDisplay(); autoClick();
