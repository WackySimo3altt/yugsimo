body { background-color: black; color: white; display: flex; justify-content: center; align-items: center; height: 100vh; margin: 0; font-family: Arial, sans-serif; } .container { text-align: center; } #clicker-tile { background-color: #444; border: 2px solid #fff; padding: 20px; cursor: pointer; user-select: none; margin-bottom: 10px; } #clicker-tile:hover { background-color: #555; } #click-count, #upgrades, #rebirths, #level { margin: 10px 0; } let clickCount = 0; let upgrades = 0; let rebirths = 0; let level = 1; let autoClickerInterval = null; function incrementClicks(amount = 1) { clickCount += amount; document.getElementById('click-count').innerText = `${clickCount} Clicks`; checkUpgrades(); checkRebirth(); checkLevelUp(); } function startAutoClicker() { if (!autoClickerInterval) { autoClickerInterval = setInterval(() => incrementClicks(20), 50); } } function stopAutoClicker() { if (autoClickerInterval) { clearInterval(autoClickerInterval); autoClickerInterval = null; } } function checkUpgrades() { if (clickCount >= 100000 * (upgrades + 1)) { upgrades++; document.getElementById('upgrades').innerText = `Upgrades: ${upgrades}`; } } function checkRebirth() { if (clickCount >= 100000) { clickCount = 0; upgrades = 0; rebirths++; document.getElementById('click-count').innerText = `${clickCount} Clicks`; document.getElementById('rebirths').innerText = `Rebirths: ${rebirths}`; document.getElementById('upgrades').innerText = 'Upgrades: None'; } } function checkLevelUp() { if (clickCount >= 1000000 * level) { level++; document.getElementById('level').innerText = `Level: ${level}`; } } document.addEventListener('keydown', (event) => { if (event.code === 'Space') { startAutoClicker(); } }); document.addEventListener('keyup', (event) => { if (event.code === 'Space') { stopAutoClicker(); } });
