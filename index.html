const canvas = document.getElementById('gameCanvas'); const context = canvas.getContext('2d'); const balls = []; const player = { x: canvas.width / 2, y: canvas.height / 2 }; canvas.addEventListener('click', (event) => { const angle = Math.atan2(event.clientY - player.y, event.clientX - player.x); const speed = 5; balls.push({ x: player.x, y: player.y, dx: Math.cos(angle) * speed, dy: Math.sin(angle) * speed }); }); function update() { for (let i = 0; i < balls.length; i++) { balls[i].x += balls[i].dx; balls[i].y += balls[i].dy; if (balls[i].x < 0 || balls[i].x > canvas.width || balls[i].y < 0 || balls[i].y > canvas.height) { balls.splice(i, 1); i--; } } } function draw() { context.clearRect(0, 0, canvas.width, canvas.height); context.fillStyle = 'blue'; context.fillRect(player.x - 10, player.y - 10, 20, 20); // Draw player context.fillStyle = 'red'; for (const ball of balls) { context.beginPath(); context.arc(ball.x, ball.y, 5, 0, Math.PI * 2); context.fill(); } } function gameLoop() { update(); draw(); requestAnimationFrame(gameLoop); } gameLoop();
